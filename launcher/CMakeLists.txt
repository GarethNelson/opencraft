cmake_minimum_required(VERSION 2.8.3)

cmake_policy(SET CMP0004 OLD)

list(APPEND CMAKE_MODULE_PATH "${CMAKE_CURRENT_LIST_DIR}/CMakeFiles")


FIND_PACKAGE(Boost COMPONENTS system)

find_package(SDL2 REQUIRED)
INCLUDE(FindPkgConfig)

set (CMAKE_CXX_STANDARD 11)

include(CheckCXXCompilerFlag)
CHECK_CXX_COMPILER_FLAG("-std=c++11" COMPILER_SUPPORTS_CXX11)
CHECK_CXX_COMPILER_FLAG("-std=c++0x" COMPILER_SUPPORTS_CXX0X)
if(COMPILER_SUPPORTS_CXX11)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11")
elseif(COMPILER_SUPPORTS_CXX0X)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++0x")
else()
        message(STATUS "The compiler ${CMAKE_CXX_COMPILER} has no C++11 support. Please use a different C++ compiler.")
endif()

set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wno-narrowing")

INCLUDE_DIRECTORIES(${Boost_INCLUDE_DIRS} ${SDL2_INCLUDE_DIRS} ${CMAKE_SOURCE_DIR}/include )

FILE(GLOB launcher_src     ${CMAKE_SOURCE_DIR}/src/*.cpp)

ADD_EXECUTABLE(launcher ${soil_src} ${oglconsole_src} ${launcher_src})

if(${CMAKE_SYSTEM_NAME} STREQUAL "Linux")
   set(EXTRA_GL GL GLEW GLU)
endif()


TARGET_LINK_LIBRARIES(launcher ${EXTRA_GL} ${SDL2_LIBRARIES} ${Boost_LIBRARIES})

ADD_CUSTOM_COMMAND( OUTPUT ${CMAKE_SOURCE_DIR}/include/version.h
                    COMMAND ${CMAKE_SOURCE_DIR}/autogen.sh
                    WORKING_DIRECTORY ${CMAKE_SOURCE_DIR}
                    COMMENT "Creating autogenerated headers" )

add_custom_target(autogen_headers
                  DEPENDS ${CMAKE_SOURCE_DIR}/include/version.h)

add_dependencies(launcher autogen_headers)
